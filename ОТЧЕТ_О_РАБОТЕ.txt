                    ИНСТРУКЦИЯ ПО ЗАЛИВКЕ НА СЕРВЕР
================================================================================

ШАГ 1: ПОДГОТОВКА СЕРВЕРА
-------------------------
# Обновление системы
sudo apt update && sudo apt upgrade -y

# Установка необходимых пакетов
sudo apt install nginx php-fpm php-cli supervisor certbot python3-certbot-nginx -y

ШАГ 2: ЗАГРУЗКА ПРОЕКТА
-----------------------
# Создание директории
sudo mkdir -p /var/www/biomix.uz
sudo chown -R www-data:www-data /var/www/biomix.uz

# Загрузка файлов (выберите один из способов):
# Способ 1: Через Git
sudo git clone https://github.com/Ibrakam/Biomix.git /var/www/biomix.uz

# Способ 2: Через SCP
scp -r /Users/ibragimkadamzanov/Downloads/Backup\ 19-10-20/* user@your-server:/var/www/biomix.uz/

ШАГ 3: НАСТРОЙКА КОНФИГУРАЦИЙ
-----------------------------
# Копирование конфигурационных файлов
sudo cp /var/www/biomix.uz/biomix.conf /etc/php/8.1/fpm/pool.d/
sudo cp /var/www/biomix.uz/biomix.nginx.conf /etc/nginx/sites-available/biomix.uz
sudo cp /var/www/biomix.uz/biomix-supervisor.conf /etc/supervisor/conf.d/

# Активация сайта
sudo ln -s /etc/nginx/sites-available/biomix.uz /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx

ШАГ 4: НАСТРОЙКА PHP И ПРАВ ДОСТУПА
-----------------------------------
# Перезапуск PHP-FPM
sudo systemctl restart php8.1-fpm

# Настройка прав доступа
sudo chown -R www-data:www-data /var/www/biomix.uz
sudo chmod -R 755 /var/www/biomix.uz

ШАГ 5: НАСТРОЙКА SUPERVISOR
---------------------------
# Перезагрузка Supervisor
sudo supervisorctl reread
sudo supervisorctl update
sudo supervisorctl start biomix-php-fpm
sudo supervisorctl start biomix-nginx

ШАГ 6: НАСТРОЙКА SSL СЕРТИФИКАТА
--------------------------------
# Получение SSL сертификата
sudo certbot --nginx -d biomix.uz -d www.biomix.uz

# Автоматическое обновление
sudo crontab -e
# Добавить: 0 12 * * * /usr/bin/certbot renew --quiet

ШАГ 7: НАСТРОЙКА DNS
--------------------
В панели управления доменом добавить A-записи:
biomix.uz     A    YOUR_SERVER_IP
www.biomix.uz A    YOUR_SERVER_IP
